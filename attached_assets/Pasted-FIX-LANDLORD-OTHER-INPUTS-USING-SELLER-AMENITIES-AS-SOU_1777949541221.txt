Universal handler allowed:
If it is safer, create one reusable helper like:

function handleOtherToggle(selectId, wrapperId) {
    const $select = $('#' + selectId);
    const $wrapper = $('#' + wrapperId);

    if (!$select.length || !$wrapper.length) return;

    function toggle() {
        const val = $select.val();
        const show =
            (Array.isArray(val) && val.includes('Other')) ||
            val === 'Other';

        if (show) {
            $wrapper.show();
        } else {
            $wrapper.hide();
        }
    }

    toggle();
    $select.off('change.otherToggle').on('change.otherToggle', toggle);
}
