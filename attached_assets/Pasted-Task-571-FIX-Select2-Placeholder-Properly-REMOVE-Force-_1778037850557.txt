STEP 2 — FIX INITIALIZATION (THE REAL FIX)

Find ALL Select2 init code.

REPLACE ALL of them with this SINGLE initializer:

function initSelect2Properly() {
    $('.select2-multiple, .select2-single').each(function () {
        const $el = $(this);

        // Destroy broken existing init
        if ($el.hasClass('select2-hidden-accessible')) {
            $el.select2('destroy');
        }

        const placeholder = $el.attr('data-placeholder') || 'Select';

        $el.select2({
            placeholder: placeholder,
            allowClear: true,
            width: '100%'
        });
    });
}

    });
