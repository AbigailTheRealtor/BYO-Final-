ROOT CAUSE:
Some Select2 fields are already initialized by older/broken custom initializers, so the new initSelect2Fields() skips them. The rendered Select2 DOM never receives visible placeholder text.

ADD THIS FUNCTION TO EACH MAIN CREATE BLADE FILE AFTER EXISTING SELECT2 INIT CODE:

function forceSelect2Placeholders() {
    $('.select2-multiple, .select2-single').each(function () {
        var $el = $(this);
        var placeholder = $el.attr('data-placeholder') || 'Select';
        var $container = $el.next('.select2-container');

        if (!$container.length) {
            return;
        }

        // Multi-select placeholder repair
        var $searchField = $container.find('.select2-search__field');
        if ($searchField.length && (!$el.val() || $el.val().length === 0)) {
            $searchField.attr('placeholder', placeholder);
            $searchField.css({
                'width': '100%',
                'min-width': '120px'
            });
        }

        // Single-select placeholder repair
        var $rendered = $container.find('.select2-selection__rendered');
        if ($rendered.length && (!$el.val() || $el.val() === '')) {
            $rendered.text(placeholder);
            $rendered.addClass('select2-selection__placeholder');
        }
    });
}

RUN IT AFTER ALL EXISTING INIT CODE:

$(document).ready(function () {
    setTimeout(forceSelect2Placeholders, 300);
});

document.addEventListener('livewire:load', function () {
    Livewire.hook('message.processed', () => {
        setTimeout(forceSelect2Placeholders, 300);
    });
});
