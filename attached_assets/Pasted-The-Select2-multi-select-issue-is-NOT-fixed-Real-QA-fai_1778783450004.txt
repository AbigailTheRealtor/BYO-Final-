Create or enforce ONE shared helper for true multi-selects in the active Hire Agent component:

function initTrueMultiSelect($el) {
    if (!$el.length) return;

    if ($el.hasClass('select2-hidden-accessible')) {
        $el.select2('destroy');
    }

    $el.select2({
        placeholder: $el.data('placeholder') || 'Select',
        allowClear: true,
        width: '100%',
        closeOnSelect: false
    });
}
