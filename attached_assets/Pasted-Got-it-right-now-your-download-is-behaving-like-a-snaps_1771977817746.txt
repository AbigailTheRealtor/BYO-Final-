# STEP 3 — Add a shared Export Formatter (handles arrays, money, dates, “Other”)

Create:

`app/Exports/ListingExportFormatter.php`

```php
<?php

namespace App\Exports;

use Illuminate\Support\Arr;

class ListingExportFormatter
{
    public static function normalizePropertyType(?string $v): ?string
    {
        if (!$v) return $v;
        $v = trim($v);
        // Display-only cleanup:
        $v = preg_replace('/\s+Property$/i', '', $v);
        return $v;
    }

    public static function toText($value): string
    {
        if (is_null($value)) return '';
        if (is_bool($value)) return $value ? 'Yes' : 'No';
        if (is_array($value)) {
            $flat = self::flatten($value);
            return implode(', ', array_values(array_unique(array_filter(array_map('trim', $flat)))));
        }
        return trim((string)$value);
    }

    public static function toList($value): array
    {
        if (is_null($value)) return [];
        if (is_string($value)) {
            $decoded = json_decode($value, true);
            if (json_last_error() === JSON_ERROR_NONE) $value = $decoded;
        }
        if (is_array($value)) return array_values(array_filter(array_map('trim', self::flatten($value))));
        $s = trim((string)$value);
        return $s === '' ? [] : [$s];
    }

    public static function flatten(array $arr): array
    {
        $out = [];
        foreach ($arr as $k => $v) {
            if (is_array($v)) $out = array_merge($out, self::flatten($v));
            else $out[] = (string)$v;
        }
        return $out;
    }

    /**
     * "Other" resolution:
     * - If the selection is "Other" (or includes Other in multi), and custom field is present,
     *   use the custom text instead of the literal "Other".
     * - If both exist, replace Other with custom text.
     */
    public static function resolveOther($selectionValue, $otherTextValue): array
    {
        $items = self::toList($selectionValue);
        $otherText = trim((string)($otherTextValue ?? ''));

        if ($otherText === '') return $items;

        $result = [];
        foreach ($items as $it) {
            if (strcasecmp($it, 'Other') === 0 || str_starts_with(strtolower($it), 'other')) {
                $result[] = $otherText;
            } else {
                $result[] = $it;
            }
        }

        // If selection was exactly "Other" and got removed, ensure custom appears
        if (count($result) === 0) $result[] = $otherText;

        return array_values(array_unique(array_filter(array_map('trim', $result))));
    }
}
```

# STEP 4 — Build a unified PDF renderer that uses the field map

Create:

`app/Exports/ListingPdfDataBuilder.php`

```php
<?php

namespace App\Exports;

class ListingPdfDataBuilder
{
    public static function build($listing, array $sections, array $otherPairs = [], array $normalizers = []): array
    {
        $out = [];

        foreach ($sections as $sectionTitle => $fields) {
            $rows = [];

            foreach ($fields as $label => $key) {
                $value = data_get($listing, $key);

                // Apply field-specific normalizers (optional)
                if (isset($normalizers[$key]) && is_callable($normalizers[$key])) {
                    $value = $normalizers[$key]($value);
                }

                // Resolve "Other" (if mapped)
                if (isset($otherPairs[$key])) {
                    $otherValue = data_get($listing, $otherPairs[$key]);
                    $items = ListingExportFormatter::resolveOther($value, $otherValue);
                } else {
                    $items = ListingExportFormatter::toList($value);
                }

                // Skip empty fields (or show blank if you prefer)
                if (count($items) === 0) continue;

                $rows[] = [
                    'label' => $label,
                    'items' => $items,
                ];
            }

            if (count($rows)) {
                $out[] = [
                    'title' => $sectionTitle,
                    'rows' => $rows,
                ];
            }
        }

        return $out;
    }
}
```
