```js
function buyerForceRestoreSelect2(idOrClass, lwKey, isClass = false) {
    try {
        let $el = isClass ? $(idOrClass) : $(idOrClass);
        if (!$el.length) return;

        let val = @this.get(lwKey);

        // Normalize to array for multi-selects
        if (val === null || typeof val === 'undefined') val = [];
        if (!Array.isArray(val)) {
            try { val = JSON.parse(val); } catch(e) { val = [val].filter(Boolean); }
        }

        // If class selector, apply to all
        if (isClass) {
            $el.each(function () {
                $(this).val(val).trigger('change.select2');
            });
        } else {
            $el.val(val).trigger('change.select2');
        }

        console.log('[BUYER RESTORE]', lwKey, val);
    } catch (e) {
        console.log('[BUYER RESTORE ERROR]', lwKey, e);
    }
}

function buyerBindSelect2Sync($el, lwKey) {
    // Namespaced to avoid stacking
    $el.off('change.buyerSync').on('change.buyerSync', function () {
        let selected = $el.val() || [];
        @this.set(lwKey, selected);
        console.log('[BUYER SET]', lwKey, selected);
    });
}
```

### Step C1: Right before saving meta, normalize property_items + other_property_items

Add this logic right before you call `saveMeta('property_items', ...)`:

```php
// Normalize "Other" token for property_items
$items = $this->property_items ?? [];
if (is_string($items)) {
    $decoded = json_decode($items, true);
    $items = is_array($decoded) ? $decoded : [];
}
$items = array_values(array_filter($items));

$otherText = trim((string)($this->other_property_items ?? ''));

if ($otherText !== '' && !in_array('Other', $items)) {
    $items[] = 'Other';
}
if ($otherText === '' && in_array('Other', $items)) {
    $items = array_values(array_diff($items, ['Other']));
}

$this->property_items = $items;
```

Then save meta as normal.
