    $('.select2-multiple, .select2-single').each(function () {
        var $el = $(this);

        // ALWAYS destroy first if already initialized
        if ($el.hasClass('select2-hidden-accessible')) {
            $el.select2('destroy');
        }

        var placeholder = $el.attr('data-placeholder') || 'Select';

        $el.select2({
            placeholder: placeholder,
            allowClear: true,
            width: '100%'
        });
    });
}
