2. Add ONE top-level function in each file, not nested inside any other function:

function initSelect2Fields() {
    $('.select2-multiple, .select2-single').each(function () {
        var $el = $(this);

        if ($el.hasClass('select2-hidden-accessible')) {
            $el.select2('destroy');
        }

        var placeholder = $el.attr('data-placeholder') || 'Select';

        $el.select2({
            placeholder: placeholder,
            allowClear: true,
            width: '100%'
        });
    });
}

3. Ensure existing calls run this function:
- on document ready
- after Livewire message.processed

Use:

    initSelect2Fields();
});

document.addEventListener('livewire:load', function () {
    Livewire.hook('message.processed', function () {
        initSelect2Fields();
    });
});
